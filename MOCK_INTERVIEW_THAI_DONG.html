Tại sao chọn Go?

  • Goroutines: Mỗi exam session được xử lý trong 1 goroutine riêng. Với 5.000 thí sinh đồng thời, Go xử lý nhẹ nhàng — goroutine chỉ tốn ~2 KB stack ban đầu, so với ~1 MB của thread Java
  • Low latency: Anti-cheat cần xử lý event trong vòng milliseconds. Go compile thành binary, không có overhead của JVM hay PHP interpreter
  • Concurrency model: Channel + goroutine pattern rất phù hợp với event processing pipeline

Tại sao Redis làm buffer thay vì push thẳng vào Kafka?

  • Tần suất cực cao: Mỗi thí sinh có thể gửi 10–50 events/phút (mouse movement, tab switch, keystroke pattern). Với 5.000 thí sinh = 50K–250K events/phút. Push thẳng vào Kafka tạo quá nhiều small messages, không hiệu quả
  • Batching: Redis buffer cho phép gom events theo exam session, flush mỗi 1–2 giây để analyze theo batch — giảm 95% message volume lên Kafka
  • Tốc độ: Redis LPUSH là O(1), latency ~0.1 ms. Kafka produce ~5–10 ms. Với high-frequency events, sự khác biệt này rất quan trọng
  • Transient data: Raw cheat events chỉ cần giữ vài phút để analyze. Không cần Kafka's durability cho data này

type AntiCheatProcessor struct {
    redisClient *redis.Client
    kafkaWriter *kafka.Writer
    rules       []DetectionRule
}

type CandidateEvent struct {
    ExamID      string                 `json:"exam_id"`
    CandidateID string                 `json:"candidate_id"`
    EventType   string                 `json:"event_type"` // tab_switch, focus_loss, rapid_submit
    Timestamp   time.Time              `json:"timestamp"`
    Metadata    map[string]interface{} `json:"metadata"`
}

// Buffer event vào Redis — gọi cho mỗi event đến
func (p *AntiCheatProcessor) BufferEvent(ctx context.Context, event CandidateEvent) error {
    key := fmt.Sprintf("cheat:events:%s:%s", event.ExamID, event.CandidateID)
    data, _ := json.Marshal(event)

    pipe := p.redisClient.Pipeline()
    pipe.LPush(ctx, key, data)
    pipe.Expire(ctx, key, 5*time.Minute) // TTL — tự dọn dẹp
    _, err := pipe.Exec(ctx)
    return err
}

// Flush & analyze — chạy mỗi 1 giây cho mỗi exam đang active
func (p *AntiCheatProcessor) FlushAndAnalyze(ctx context.Context, examID, candidateID string) {
    key := fmt.Sprintf("cheat:events:%s:%s", examID, candidateID)

    // Atomic: lấy toàn bộ events và xóa buffer
    rawEvents, err := p.redisClient.LRange(ctx, key, 0, -1).Result()
    if err != nil || len(rawEvents) == 0 {
        return
    }
    p.redisClient.Del(ctx, key)

    // Parse & apply detection rules
    var events []CandidateEvent
    for _, raw := range rawEvents {
        var e CandidateEvent
        json.Unmarshal([]byte(raw), &e)
        events = append(events, e)
    }

    for _, rule := range p.rules {
        if violation := rule.Evaluate(events); violation != nil {
            p.publishViolation(ctx, violation)
        }
    }
}

// Detection rule: Tab switch frequency
type TabSwitchRule struct {
    MaxCount int
    Window   time.Duration
}

func (r *TabSwitchRule) Evaluate(events []CandidateEvent) *Violation {
    count := 0
    cutoff := time.Now().Add(-r.Window)

    for _, e := range events {
        if e.EventType == "tab_switch" && e.Timestamp.After(cutoff) {
            count++
        }
    }

    if count >= r.MaxCount {
        return &Violation{
            Type:     "FREQUENT_TAB_SWITCH",
            Severity: "WARNING",
            Message:  fmt.Sprintf("%d lần tab switch trong %v", count, r.Window),
        }
    }
    return nil
}

// Chỉ publish violations lên Kafka — giảm 95% message volume
func (p *AntiCheatProcessor) publishViolation(ctx context.Context, v *Violation) {
    data, _ := json.Marshal(v)
    p.kafkaWriter.WriteMessages(ctx, kafka.Message{
        Topic: "cheat.detected",
        Key:   []byte(v.CandidateID),
        Value: data,
    })
}

  • Redis buffer hấp thu spike traffic, tránh overwhelm downstream
  • Rules engine có thể thêm/sửa rule mà không cần redeploy service
  • Chỉ violations mới được push lên Kafka — giảm 95% message volume so với push raw events
  • Mỗi exam session được isolate — 1 session có vấn đề không ảnh hưởng session khác

func (h *ExamHandler) SubmitExam(w http.ResponseWriter, r *http.Request) {
    var req SubmitExamRequest
    json.NewDecoder(r.Body).Decode(&req)

    ctx := r.Context()
    candidateID := ctx.Value("candidateID").(string)

    // 1. Validate session
    session, err := h.examRepo.GetActiveSession(ctx, req.ExamID, candidateID)
    if err != nil {
        http.Error(w, "Phòng thi không tồn tại", http.StatusNotFound)
        return
    }
    if time.Now().After(session.EndsAt) {
        http.Error(w, "Hết giờ thi", http.StatusForbidden)
        return
    }

    // 2. Persist answers — dùng UPSERT để idempotent
    submission := &ExamSubmission{
        ID:          uuid.New().String(),
        ExamID:      req.ExamID,
        CandidateID: candidateID,
        Answers:     req.Answers,
        Status:      "PENDING_SCORE",
        SubmittedAt: time.Now(),
    }
    h.examRepo.UpsertSubmission(ctx, submission) // ON CONFLICT DO UPDATE

    // 3. Publish async scoring event
    event := ExamSubmittedEvent{
        SubmissionID: submission.ID,
        ExamID:       req.ExamID,
        CandidateID:  candidateID,
        Timestamp:    time.Now(),
    }
    if err := h.kafkaProducer.Publish(ctx, "exam.submitted", submission.ID, event); err != nil {
        // Nếu Kafka fail, reconciliation job sẽ retry sau
        log.Error("Không publish được event", "error", err)
    }

    // 4. Trả về ngay — scoring xảy ra async
    json.NewEncoder(w).Encode(map[string]string{
        "status":        "SUBMITTED",
        "submission_id": submission.ID,
        "message":       "Bài thi đã được nhận. Kết quả sẽ có trong ít phút.",
    })
}

🛡️ 3 Lớp bảo vệ để không mất bài thi

Lớp 1 — Persist TRƯỚC khi publish event: Answers được lưu vào DB trước (status: PENDING_SCORE). Nếu Kafka down, bài thi vẫn còn trong DB.

Lớp 2 — Idempotent submission (Upsert): Thí sinh có thể bấm "Nộp bài" nhiều lần (network lag, double click). Dùng ON CONFLICT (exam_id, candidate_id) DO UPDATE — chỉ giữ bản mới nhất.

Learning Profile là một read-optimized aggregate được xây từ nhiều event sources:

// "exam.submitted"       → Cập nhật số bài thi đã làm

// "lesson.completed"     → Cập nhật tiến độ học tập

// Partition key = user_id → events của cùng 1 user được xử lý theo đúng thứ tự

func (s *ProfileService) OnScoreCalculated(ctx context.Context, event ScoreCalculatedEvent) error {
    profile, _ := s.repo.GetOrCreate(ctx, event.CandidateID)

    // Idempotency: kiểm tra bài thi này đã được tính chưa
    if s.repo.IsExamAlreadyCounted(ctx, event.CandidateID, event.ExamID) {
        return nil // Đã xử lý rồi, bỏ qua
    }

    // Cập nhật aggregates
    profile.TotalExams++
    profile.AverageScore = recalculateAverage(
        profile.AverageScore, profile.TotalExams-1, event.Score,
    )

    // Cập nhật skill breakdown
    for skill, score := range event.SkillScores {
        profile.SkillBreakdown[skill] = recalculateAverage(
            profile.SkillBreakdown[skill], profile.TotalExams-1, score,
        )
    }

    profile.LastActivityAt = event.Timestamp
    return s.repo.Save(ctx, profile)
}

Chiến lược đảm bảo consistency:

  • Idempotency: Mỗi event có unique ID. Trước khi apply, kiểm tra event này đã được xử lý chưa
  • Ordering: Kafka đảm bảo ordering per partition. Partition by user_id nên events của cùng 1 user được xử lý theo đúng thứ tự
  • Rebuild capability: Vì dùng events, có thể replay từ Kafka để rebuild toàn bộ profile nếu cần
⚠️ Trade-off em chấp nhận:
  • Profile không real-time 100% (delay 1–5 giây) nhưng OK cho learning analytics — người dùng không cần xem điểm trung bình ngay lập tức
  • Nếu consumer bị lag, profile tạm thời cũ — nhưng sẽ tự catch up khi consumer xử lý xong hàng đợi
  • Storage tăng vì giữ cả events lẫn aggregate — nhưng cho phép audit và rebuild

public final class Shift {
    private final LocalTime startTime;
    private final LocalTime endTime;
    private final ShiftType type; // MORNING, AFTERNOON, NIGHT

    public Shift(LocalTime startTime, LocalTime endTime, ShiftType type) {
        if (endTime.isBefore(startTime) && type != ShiftType.NIGHT) {
            throw new InvalidShiftException("Giờ kết thúc phải sau giờ bắt đầu");
        }
        this.startTime = startTime;
        this.endTime   = endTime;
        this.type      = type;
    }

    public Duration getDuration() {
        if (type == ShiftType.NIGHT) {
            return Duration.between(startTime, LocalTime.MAX)
                .plus(Duration.between(LocalTime.MIN, endTime));
        }
        return Duration.between(startTime, endTime);
    }
}

// FIX 1: Quy ước thứ tự lock nhất quán
// Rule: LUÔN lock theo thứ tự: leave_requests TRƯỚC, employee_schedules SAU
// Mọi transaction phải tuân theo thứ tự này

@Service
public class LeaveApprovalService {

    @Transactional
    @Retryable(value = DeadlockException.class, maxAttempts = 3,
               backoff = @Backoff(delay = 100, multiplier = 2))
    public void approveLeave(Long leaveRequestId) {

        // Bước 1: Lock leave_requests TRƯỚC (consistent order)
        LeaveRequest leave = leaveRepo.findByIdWithLock(leaveRequestId);

        // Bước 2: Sau đó mới lock schedule
        EmployeeSchedule schedule = scheduleRepo
            .findByEmployeeAndDateWithLock(leave.getEmployeeId(), leave.getDate());

        leave.setStatus(LeaveStatus.APPROVED);
        schedule.setDayOff(true);

        leaveRepo.save(leave);
        scheduleRepo.save(schedule);
    }
}

@Service
public class ShiftAssignmentService {

    @Transactional
    @Retryable(value = DeadlockException.class, maxAttempts = 3,
               backoff = @Backoff(delay = 100, multiplier = 2))
    public void assignShift(Long employeeId, LocalDate date, Long shiftId) {

        // Bước 1: Lock leave_requests TRƯỚC — CÙNG THỨ TỰ với trên!
        leaveRepo.findPendingByEmployeeAndDateWithLock(employeeId, date)

    total_capacity  INT NOT NULL,           -- 500 (admin set, ít thay đổi)

    booked          INT NOT NULL DEFAULT 0, -- đã thanh toán xong

-- Cancel booking đã confirm → Release:
    booked -= 5         -- available tăng lại 5

🎯 Chống Oversell: Redis Pre-check + DB Optimistic Lock

Vấn đề race condition:

-- Còn 6 chỗ. Cùng lúc: Khách A đặt 5 vé, Khách B đặt 3 vé
Thread A: READ available=6 → check 5≤6 ✓ → UPDATE reserved+=5
Thread B: READ available=6 → check 3≤6 ✓ → UPDATE reserved+=3
                ↑ đọc cùng giá trị cũ vì A chưa commit
-- Kết quả: bán 8 vé cho 6 chỗ → OVERSELL!

Solution: Redis Lua (reject sớm) + DB Optimistic Lock (source of truth):

-- Redis Lua script — atomic pre-check (không thể bị race condition)
local key      = KEYS[1]  -- "inventory:{park}:{type}:{date}"
local qty      = tonumber(ARGV[1])
local total    = tonumber(redis.call('HGET', key, 'total')    or 0)
local reserved = tonumber(redis.call('HGET', key, 'reserved') or 0)
local booked   = tonumber(redis.call('HGET', key, 'booked')   or 0)
local available = total - reserved - booked

if available < qty then
    return -1  -- hết chỗ, reject ngay tại Redis
end

redis.call('HINCRBY', key, 'reserved', qty)
return available - qty  -- remaining sau khi reserve
-- DB Optimistic Lock — source of truth
-- Nếu version thay đổi giữa lúc đọc và ghi → conflict → retry
UPDATE daily_inventory
SET    reserved = reserved + :quantity,
       version  = version  + 1
WHERE  park_id        = :parkId
  AND  ticket_type_id = :ticketTypeId
  AND  visit_date     = :visitDate
  AND  version        = :expectedVersion          -- optimistic check
  AND  (total_capacity - reserved - booked) >= :quantity;  -- safety check
-- affected rows = 0 → conflict → rollback Redis, retry (max 3 lần)

Khi DB fail → Rollback Redis:

// Compensation: hoàn lại Redis counter nếu DB conflict
if (affectedRows == 0) {

Quy tắc quan trọng của State Machine:

  • EXPIRED và USED là terminal states — không bao giờ revert
  • Payment FAILED ≠ EXPIRED: Payment fail → booking vẫn PENDING, cho phép user retry bằng phương thức khác (max 3 lần). Chỉ expire khi hết TTL
  • CONFIRMED → CANCELLED phải trigger refund async, release booked inventory
  • Price snapshot: Giá được snapshot vào booking record tại thời điểm tạo. Payment dùng giá này, không query giá mới — tránh "admin đổi giá giữa chừng"
💡 Tại sao cho retry thay vì cancel ngay khi payment fail?
  • Cancel ngay → user phải đặt lại từ đầu, inventory release → có thể bị người khác mua mất
  • Cho retry (max 3, trong TTL) → user chỉ cần chọn phương thức thanh toán khác, inventory vẫn giữ cho họ
  • TTL vẫn chạy → không giữ chỗ mãi nếu user bỏ đi

    } else {
        expirePayment(p); // chỉ expire khi chắc chắn gateway không charge
    }
}

Idempotency của handler — gateway retry callback nhiều lần:

public void handleWebhookCallback(GatewayCallback callback) {
    Payment payment = paymentRepo.findByGatewayTxnId(callback.getTxnId());

    // Idempotent: đã SUCCESS rồi → skip
    if (payment.getStatus() == SUCCESS) {
        return; // trả 200 OK để gateway không retry nữa
    }

    // Chỉ PENDING → SUCCESS là transition hợp lệ
    if (payment.getStatus() != PENDING) {
        throw new InvalidStateTransitionException();
    }

    // Verify amount không khớp → reject (chống tampering)
    if (!callback.getAmount().equals(payment.getAmount())) {
        throw new AmountMismatchException();
    }

    confirmPaymentAndBooking(payment); // optimistic lock để chặn race condition
}
